def clean_currency(value):
    value = value.strip()
    if 'B' in value:
        return float(value.replace('$', '').replace('B', '').replace(',', ''))
    elif 'M' in value:
        return float(value.replace('$', '').replace('M', '').replace(',', '')) / 1000
    else:
        return float(value.replace('$', '').replace(',', ''))

def clean_currency(value):
    """Clean and convert currency values to numeric."""
    if isinstance(value, str):
        value = value.strip()
        if 'B' in value:
            return float(value.replace('$', '').replace('B', '').replace(',', ''))
        elif 'M' in value:
            return float(value.replace('$', '').replace('M', '').replace(',', '')) / 1000
    elif isinstance(value, (int, float)):
        return float(value)  # Return the number as is if it's already numeric
    return pd.NA  # Handle non-numeric values as NaN

# Step 1: Clean currency columns
df['$ Last change'] = df['$ Last change'].apply(clean_currency)
df['$ YTD change'] = df['$ YTD change'].apply(clean_currency)

# Step 2: Group by 'Industry' and calculate the total net worth and change
industry_growth = df.groupby('Industry').agg(
    total_net_worth=('Total net worth', 'sum'),
    last_year_change=('$ Last change', 'sum'),
    ytd_change=('$ YTD change', 'sum')
).dropna()  # Drop NaN values

# Step 3: Plot the results
industry_growth[['last_year_change', 'ytd_change']].plot(kind='bar', figsize=(10, 6), color=['skyblue', 'orange'])

# Adding title and labels
plt.title('Growth in Net Worth Across Industries')
plt.xlabel('Industry')
plt.ylabel('Growth in Net Worth')

plt.show()
